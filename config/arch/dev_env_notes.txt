// disable cloud.init
sudo vim /etc/cloud/cloud.init.cfg.d/..  // may not applicable

// configure static ip
sudo vim /etc/netplan/<interface>-cfg.yaml
sudo vim /etc/netplan/ens6f1-cfg.yaml  // example
network:
        version: 2
        renderer: networkd
        ethernets:
                ens6f1:
                        dhcp4: no
                        addresses:
                                - 192.168.85.83/24
                        gateway4: 192.168.85.1
                        nameservers:
                                addresses: [8.8.8.8, 1.1.1.1]
                                
// apply configurations
sudo netplan apply
ip addr show dev <interface>
ip addr show dev ens6f1

// take interfaces down | up - netplan
sudo ip link set <interface> down
sudo ip link set <interface> up

// all devices in /etc/network/interface
sudo ifdown -a
sudo ifup -a
